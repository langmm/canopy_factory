    @cached_property
    def ePhotosynthesis_vars(self):
        r"""ePhotosynthesis.Variables: Variables for the simulation."""
        import ePhotosynthesis
        from ePhotosynthesis import Variables
        out = Variables()
        if self.args.param:
            for iparam in self.args.param:
                out.readParam(iparam)
        if self.args.enzyme:
            out.readEnzymeAct(self.args.param)
        if self.args.grnc:
            out.readGRN(self.args.grnc)
        if self.args.driver in ['CM', 'EPS']:
            out.RUBISCOMETHOD = 2
        ePhotosynthesis.modules.PR.setRUBISCOTOTAL(3)
        return out

    @cached_property
    def ePhotosynthesis_driver(self):
        r"""ePhotosynthesis.driver: Driver for the simulation."""
        import ePhotosynthesis
        cls = getattr(ePhotosynthesis.drivers,
                      f"{self.args.driver}Driver")
        return cls(self.ePhotosynthesis_vars)

    def ePhotosynthesis(self, args):
        driver = ['trDynaPS', 'DynaPS', 'CM', 'EPS'].index(self.args.driver) + 1
        if self.args.driver in ['CM', 'EPS']:
            args.setdefault('RUBISCOMETHOD', 2)
        args.setdefault('PR::MOD::RUBISCOTOTAL', 3)
        kwargs = {
            'inputs': {k: str(v) for k, v in args.items()},
        }
        if self.args.param:
            kwargs['evnFile'] = self.args.param
        if self.args.enzyme:
            kwargs['enzymeFile'] = self.args.enzyme
        if self.args.grnc:
            kwargs['grnFile'] = self.args.grnc
        return ePhotosynthesis.run_simulation(driver, **kwargs)

    def run_ePhoto(self, flux_density, area, dont_convert=False):
        r"""Run ePhotosynthesis.

        Args:
            flux_density (units.Quantity): Flux density to pass to
                ePhotosynthesis as TestLi parameters. Should be in
                units compatible with umol s**-1 m**-2.
            area (units.Quantity): Area over which the flux density
                should be applied.
            dont_convert (bool, optional): If True, don't convert
                flux density to umol s**-1 m**-2 (assumes it is already
                converted).

        Returns:
            dict: Photosynthesis properties.

        """
        if dont_convert:
            flux_density_pm2 = flux_density
        else:
            flux_density_pm2 = flux_density.to('umol s**-1 m**-2')
        out = self.ePhotosynthesis({'TestLi': flux_density_pm2})
        # TODO: Add units & add CO2 assimilation rate
        pprint.pprint(out)
        pdb.set_trace()
        return out



    def _generate_output(self, name):
        r"""Generate the specified output value.

        Args:
            name (str): Name of the output to generate.

        Returns:
            object: Generated output.

        """
        if name == 'photosynthesis':
            raytrace = self.output_task('raytrace')
            if self.args.average_over == 'face':
                area = raytrace.calculate_query('areas')
                flux_density = raytrace.calculate_query('flux_density')
                flux_density_pm2 = flux_density.to('umol s**-1 m**-2')
                out = {}
                for i in range(len(area)):
                    iout = self.run_ePhoto(flux_density_pm2[i], area[i],
                                           dont_convert=True)
                    if not out:
                        out = {
                            k: np.empty((len(area), ), dtype='float64')
                            for k in iout.keys()
                        }
                    for k, v in iout.items():
                        out[k][i] = v
            elif self.args.average_over == 'plant':
                area = raytrace.calculate_query(
                    'total_areas', per_plant=True)
                flux = raytrace.calculate_query(
                    'total_flux', per_plant=True)
                out = {}
                for plantid in area.keys():
                    if plantid == 'total':
                        continue
                    iout = self.run_ePhoto(
                        flux[plantid] / area[plantid], area[plantid],
                    )
                    if not out:
                        out = {k: {} for k in iout.keys()}
                    for k, v in iout.items():
                        out[k][plantid] = v
            else:
                assert self.args.average_over == 'canopy'
                area = raytrace.calculate_query('total_areas')
                flux = raytrace.calculate_query('total_flux')
                return self.run_ePhoto(flux / area, area)
        super(PhotosynthesisTask, self)._generate_output(name)
